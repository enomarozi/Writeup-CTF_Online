import numpy as np
import matplotlib.pyplot as plt

img=plt.imread('pixel_194.png')

R=img[:,:,0]
G=img[:,:,1]
B=img[:,:,2]

ToShow=np.zeros_like(G[0:11,:])
for i in range(20):
    j=11*i
    patch = G[0+j:11+j,:]-B[0+j:11+j,:]
    if patch.shape != ToShow.shape:
        continue
    ToShow += patch

plt.imshow(ToShow)
plt.show()
